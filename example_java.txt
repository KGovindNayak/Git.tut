	{
	    int t = p;
	    int r = 0;

		        r=r*10+rem;
		        t=t/10;
	    }
	    if(p==r)
	    {
	        return true;
	    }
	    else 
	    {
	        return false;
	    }
	    
	}

class TwoDigitsSum
{
	static int getSumOfDigits(int number) 
	{
	    if(number < 0)
	    {
	        return -3;
	    }
	    else if(number > 99)
	    {
	        return -2;
	    }
	    else if(number >= 0 && number <= 9)
	    {
	        return -1;
	    }
	    else
	    {
	        int l=number/10;
	        int f=number%10;
	        return l + f;
	    }
	}
}

class RoundedSum {
	static int sumOfRoundedValues(int a, int b, int c) {
	    int ak=0,bl=0,cm=0;
	    if(a<0||b<0||c<0)
	    {
	        return -1;
	    }
	    else
	    {
	        int k=a/10;
	        if(k<5){
	           ak=k*10;
	        }
	        else{
	            ak=(k+1)*10;
	        }
	        int l=b/10;
	        if(l<5){
	            bl=l*10;
	        }
	        else{
	            bl=(l+1)*10;
	        }
	        int m=c/10;
	        if(m<5){
	             cm=m*10;
	        }
	        else{
	            cm=(m+1)*10;
	    }
	    return ak+bl+cm;
	}
}
}
